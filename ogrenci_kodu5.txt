<script>
/* =======================================
   YOL AYRIMI - Karar Verme (If / Else)
   =======================================
   Bilgisayarlar mantıklı seçimler yapmak
   için If (Eğer) ve Else (Değilse) kullanır.
   
   KURAL: "Eğer Elmas Sayısı 50'den büyükse (> 50) SAĞDAN GİT, Değilse SOLDAN GİT"
*/

// GÖREV: Şövalyenin elmas sayısını değiştirerek hangi yoldan
// gideceğini gözlemle. Ne yaparsan doğru yoldan (SAĞ)'dan gider?
let oyuncununElmasSayisi = 45; // Test etmek için bu sayıyı 60 yap

// Karar Verme Bloğu
let secilenYol = "";

if (oyuncununElmasSayisi > 50) {
    // EĞER Şart Doğruysa
    secilenYol = "SAĞ"; 
} else {
    // Şart Yanlışsa (DEĞİLSE)
    secilenYol = "SOL";
}

// ======== DEĞERLENDİRME VE GÖRSELLEŞTİRME BÖLÜMÜ ========
let panel = document.getElementById("mesaj");
let ikon = document.getElementById("yolIkonu");

// Görseli yol kuralına göre boyuyoruz:
if (secilenYol === "SAĞ") {
    document.body.style.background = "#2ecc71"; // Doğruysa Yeşil
    ikon.innerHTML = "👉";
    panel.innerHTML = "Elmas yeterli! (<span style='color:black'>" + oyuncununElmasSayisi + "</span>) SAĞDAN Gidiliyor...";
} else {
    document.body.style.background = "#e74c3c"; // Yanlışsa Kırmızı
    ikon.innerHTML = "👈";
    panel.innerHTML = "Elmas yetersiz! (<span style='color:black'>" + oyuncununElmasSayisi + "</span>) SOLDAN Gidiliyor...";
}
</script>
