<script>
/* =======================================
   BİLGE BAYKUŞ - Algoritma Takibi
   =======================================
   GÖREV: Aşağıdaki değişkenlerin değerlerini bul,
   ve ekrandaki sonucu bilgisayar hesaplasın!
*/

// PROBLEM: Markette fındığın "700" gramı "21" TL ise, "1000" gramı kaç TL eder?
// (Formül: (21 / 700) * 1000)

// 1. Önce değerlerimizi "değişken" (variable) olarak tanımlıyoruz:
let toplamFiyat = 21;
let alinanGram = 700;

let hedefGram = 1000; // Bulmak istediğimiz miktar

// 2. Hesabımızı Matematik ile Değil, Bilgisayar ile yaptıralım!
// formülünü kendin değiştir, çarp (*) böl (/)
let birimFiyat = toplamFiyat / alinanGram; // 1 gramının fiyatı
let sonucTL = birimFiyat * hedefGram; // Hedefin fiyatı

// ======== DEĞERLENDİRME VE GÖRSELLEŞTİRME BÖLÜMÜ ========
let dogruCevap = (21/700) * 1000;
let panel = document.getElementById("sonuc");

// Girdiğin formül doğruysa:
if (sonucTL === dogruCevap) {
    document.body.style.background = "#27ae60";
    panel.innerHTML = `Senin Cevabın: <span class="deger">${sonucTL} TL</span><br>Tebrikler Değerler Doğru!`;
} else {
    document.body.style.background = "#c0392b";
    panel.innerHTML = `Senin Cevabın: <span class="deger">${sonucTL || "Hata"} TL</span><br>Bir matematik hatası var!`;
}
</script>
