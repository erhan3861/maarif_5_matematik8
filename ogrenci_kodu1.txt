<script>
/* =======================================
   KADİM KAPI - Algoritma Sıralaması
   =======================================
   Buradaki listede kapıyı açmak için gereken
   adımlar var. Ancak sıraları karışmış!
   
   Görev: Dizi içindeki numaraları doğru
   mantık sırasına göre değiştir.
*/

// Adımların ID'lerini doğru sıraya koyalım.
// İPUCU: 
// 1: Cebinden çıkar 
// 2: Kilide tak
// 3: Sağa çevir
// 4: Kolu indir
// 5: Kapıyı aç
let benimSiralamam = [2, 1, 4, 3, 5]; // Buradaki sayıları değiştirip test et! (Örn: 1, 2, 3, 4, 5 yap)

// ======== DEĞERLENDİRME VE GÖRSELLEŞTİRME BÖLÜMÜ ========
const dogruSiralama = [1, 2, 3, 4, 5];
const adimYazilari = {
    1: "Anahtarı cebinden çıkar",
    2: "Anahtarı kilide tak",
    3: "Anahtarı sağa çevir",
    4: "Kapı kolunu indir",
    5: "Kapıyı iterek aç"
};

let basarili = benimSiralamam.toString() === dogruSiralama.toString();
let listeHtml = benimSiralamam.map(s => `<div class="adim">${adimYazilari[s] || "Gizemli Adım (" + s + ")"}</div>`).join("");
document.getElementById("adimlar").innerHTML = listeHtml;
let panel = document.getElementById("sonucGostergesi");
panel.style.display = "block";

if (basarili) {
    panel.style.background = "#2ecc71";
    panel.innerHTML = "🎉 DOĞRU! Kapı açıldı!";
    document.body.style.background = "#2f3640";
} else {
    panel.style.background = "#e74c3c";
    panel.innerHTML = "❌ OLMADI. Sıralama mantıksız.";
}
</script>
